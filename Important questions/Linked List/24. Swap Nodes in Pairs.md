# 24. Swap Nodes in Pairs

## 解題程式碼

```javascript
var swapPairs = function(head) {
    if (head === null || head.next === null) return head;

    const dummy = new ListNode(0, head);
    let cur = head;
    let prev = dummy;

    // 0 => 1 => 2 => 3 => 4
    // p    c    s    n

    while (cur && cur.next) {
        let second = cur.next;
        let nextPair = cur.next.next;

        // swap the pair
        second.next = cur; // 2 => 1
        cur.next = nextPair; // 1 => 3
        prev.next = second; // 0 => 2

        // update pair
        prev = cur; // prev = 1
        cur = nextPair; // cur = 3
    }
    return dummy.next;
};

// 0 => 2 => 1 => 3 => 4

// 4 => 3
// 3 => null
// 1 => 4

// 0 => 2 => 1 => 4 => 3
```
