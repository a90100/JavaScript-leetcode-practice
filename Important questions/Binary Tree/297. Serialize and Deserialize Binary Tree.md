```javascript
/**
 * Encodes a tree to a single string.
 *
 * @param {TreeNode} root
 * @return {string}
 */
var serialize = function (root) {
  const res = [];

  const DFS = (node) => {
    if (!node) {
      res.push('N');
    } else {
      res.push(node.val.toString());
      DFS(node.left);
      DFS(node.right);
    }
  };
  DFS(root);

  return res.join(',');
};

/**
 * Decodes your encoded data to tree.
 *
 * @param {string} data
 * @return {TreeNode}
 */
var deserialize = function (data) {
  const vals = data.split(',');
  let i = 0;

  const DFS = () => {
    if (vals[i] === 'N') {
      i++;
      return null;
    } else {
      const node = new TreeNode(vals[i]);
      i++;
      node.left = DFS();
      node.right = DFS();
      return node;
    }
  };

  return DFS();
};
```
