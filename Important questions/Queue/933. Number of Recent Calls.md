# 933. Number of Recent Calls

## 解題程式碼

```javascript
var RecentCounter = function () {
  this.reqs = [];
};

/**
 * @param {number} t
 * @return {number}
 */
RecentCounter.prototype.ping = function (t) {
  this.reqs.push(t);
  let preT = t - 3000;
  while (this.reqs[0] < preT) {
    this.reqs.shift();
  }
  return this.reqs.length;
};
```
